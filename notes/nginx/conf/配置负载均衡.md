```
在http里面（与server同一级），配置upstream <value> {
      server [一组服务]
  }
  
然后在proxy_pass后面，用 http://<value> 表示
```
  
 


```
http {
    include       mime.types;
    default_type  application/octet-stream;

    sendfile        on;

    keepalive_timeout  65;

    #gzip  on;
    
    upstream httpds{
        # 这里也可以写域名，weight为权重
       server 192.168.44.102:80 weight=8;
       server 192.168.44.103:80 weight=2;
       server 192.168.44.104:80 weight=1;
    }

    server {
        listen       80;
        server_name  localhost;


        location / {
            proxy_pass http://httpds;
        }

        error_page   500 502 503 504  /50x.html;
        location = /50x.html {
            root   html;
        }

    }

    server {
        listen       8080 default_server;
        listen       [::]:8080 default_server;
        server_name  _;
        include /etc/nginx/default.d/*.conf;
        client_max_body_size    1000m;
        server_tokens off;


        location / {
            root /opt/inlet;
            index index.html;
            add_header Cache-Control no-cache;
        }
        location /app1 {
            proxy_pass https://app1.test.yq.l.com/app1;
        }
        location /app2 {
            proxy_pass http://app2;
        }
    }
}
```
